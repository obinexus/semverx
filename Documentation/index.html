Rust Implementation Example

use semverx::{SemVerX, Component, VersionState};

let mut system = SemVerX::new();

// Create components with different states
let component_a = Component::new("component-a", "1.2.3")
    .with_state(VersionState::Stable);

let component_b = Component::new("component-b", "2.0.1")
    .with_state(VersionState::Experimental);

// Add dependency relationship
system.add_dependency(&component_a, &component_b);

// Perform hot-swap operation
system.hot_swap(&component_b, "2.1.0").await?;

// Resolve dependency conflicts using Eulerian cycles
let resolution = system.resolve_with_eulerian_cycles()?;
println!("Resolved dependencies: {:?}", resolution);
